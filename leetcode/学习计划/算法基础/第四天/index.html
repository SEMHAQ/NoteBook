    844. 比较含退格的字符串

    双指针

    class Solution {
    public boolean backspaceCompare(String S, String T) {
        int i = S.length() - 1, j = T.length() - 1;
        int skipS = 0, skipT = 0;

        while (i >= 0 || j >= 0) {
            while (i >= 0) {
                if (S.charAt(i) == '#') {
                    skipS++;
                    i--;
                } else if (skipS > 0) {
                    skipS--;
                    i--;
                } else {
                    break;
                }
            }
            while (j >= 0) {
                if (T.charAt(j) == '#') {
                    skipT++;
                    j--;
                } else if (skipT > 0) {
                    skipT--;
                    j--;
                } else {
                    break;
                }
            }
            if (i >= 0 && j >= 0) {
                if (S.charAt(i) != T.charAt(j)) {
                    return false;
                }
            } else {
                if (i >= 0 || j >= 0) {
                    return false;
                }
            }
            i--;
            j--;
        }
        return true;
    }
}

    986. 区间列表的交集

    双指针

    class Solution {
    public int[][] intervalIntersection(int[][] firstList, int[][] secondList) {
        int m = firstList.length,n = secondList.length;
        int i = 0, j = 0;
        List<int[]> list = new ArrayList<>();
        int[][] res;
        while(i!=m && j!=n){
            int[] first = firstList[i];
            int[] second = secondList[j];
            if(first[1]<second[0]) i++;
            else if(second[1]<first[0]) j++;
            else{//否则必有交集
                if(first[1]>=second[1]){
                    list.add(new int[]{first[0]>=second[0]?first[0]:second[0],second[1]});
                    j++;
                }
                else{
                    list.add(new int[]{first[0]>=second[0]?first[0]:second[0],first[1]});
                    i++;
                }
            }
        }
        res = new int[list.size()][2];
        for(int k = 0;k<list.size();k++){
            res[k] = list.get(k);
        }
        return res;
    }
}

    11. 盛最多水的容器

    双指针

    class Solution {
    public int maxArea(int[] height) {
                //双指针
                int l = 0,r = height.length - 1;
                int res = 0;

                while(l < r) {
                    int temp = Math.min(height[l], height[r]) * (r - l);
                    res = Math.max(temp, res);
                    if (height[l] < height[r]) {
                        l++;
                    }else {
                        r--;
                    }
                }
                return res;
    }
}
